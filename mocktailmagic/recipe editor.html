const recipes = [ { name: "Sunset Citrus Cooler", type: "citrus", ingredients: "Orange juice, lemon juice, soda water, ice", steps: "Mix juices, add ice, top with soda water, stir gently." }, { name: "Berry Bliss Fizz", type: "berry", ingredients: "Strawberries, blueberries, sparkling water, mint", steps: "Muddle berries, add ice, pour sparkling water, garnish with mint." }, { name: "Tropical Dream", type: "tropical", ingredients: "Pineapple juice, coconut water, lime, ice", steps: "Mix all ingredients, shake well, serve chilled." }, { name: "Citrus Spark Twist", type: "citrus", ingredients: "Grapefruit juice, lemon soda, orange slices", steps: "Combine juices and soda, add fruit slices for garnish." } ]; const container = document.getElementById("recipeContainer"); function displayRecipes(list) { container.innerHTML = ""; list.forEach(recipe => { const card = document.createElement("div"); card.classList.add("card"); card.innerHTML = `

${recipe.name}

${recipe.type.toUpperCase()}

`; card.onclick = () => openModal(recipe); container.appendChild(card); }); } function filterRecipes(type) { if (type === "all") { displayRecipes(recipes); } else { const filtered = recipes.filter(r => r.type === type); displayRecipes(filtered); } } function openModal(recipe) { document.getElementById("modalTitle").textContent = recipe.name; document.getElementById("modalIngredients").textContent = "Ingredients: " + recipe.ingredients; document.getElementById("modalSteps").textContent = "Steps: " + recipe.steps; document.getElementById("modal").classList.remove("hidden"); } function closeModal() { document.getElementById("modal").classList.add("hidden"); } displayRecipes(recipes);
